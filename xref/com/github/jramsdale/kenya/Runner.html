198 	private RepositorySystemSession newSession(RepositorySystem system, String repoProperty) {
199 		final File localRepositoryFile;
200 		if (repoProperty == null) {
201 			localRepositoryFile = new File(System.getProperty(USER_HOME), M2 + File.separator + REPOSITORY);
202 		} else {
203 			localRepositoryFile = new File(repoProperty);
204 			if (!localRepositoryFile.exists()) {
205 				throw new KenyaHelpException("Provided repository path [" + localRepositoryFile + "] does not exist");
206 			}
207 			if (!localRepositoryFile.isDirectory()) {
208 				throw new KenyaHelpException("Provided repository path [" + localRepositoryFile
209 						+ "] is not a directory");
210 			}
211 		}
212 		final LocalRepository localRepository = new LocalRepository(localRepositoryFile);
213 		final MavenRepositorySystemSession session = new MavenRepositorySystemSession();
214 		session.setLocalRepositoryManager(system.newLocalRepositoryManager(localRepository));
215 		return session;
216 	}

218 	private Artifact getRootArtifactTemplate(String jarPath, String gav) {
219 		if (jarPath == null && gav == null) {
220 			throw new KenyaHelpException("Must provide either " + PROJECT_NAME + ".jar or " + PROJECT_NAME
221 					+ ".gav, but not both");
222 		}
223 		if (jarPath != null) {
224 			if (gav != null) {
225 				throw new KenyaHelpException("Cannot provide both " + PROJECT_NAME + ".jar and " + PROJECT_NAME
226 						+ ".gav");
227 			} else {
228 				return createArtifactFromJar(jarPath);
229 			}
230 		}
231 		return new DefaultArtifact(gav);
232 	}

235 	private Artifact createArtifactFromJar(String jarPath) {
236 		final JarFile jarFile;
237 		try {
238 			jarFile = new JarFile(jarPath);
239 		} catch (IOException e) {
240 			throw new KenyaException("Could not load jar from file: " + jarPath, e);
241 		}
242 		for (JarEntry jarEntry : Collections.list(jarFile.entries())) {
243 			if (jarEntry.getName().startsWith("META-INF/maven/") && jarEntry.getName().endsWith("/pom.properties")) {
244 				final Properties pomProperties = new Properties();
245 				try {
246 					pomProperties.load(jarFile.getInputStream(jarEntry));
247 					return new DefaultArtifact(pomProperties.getProperty("groupId"),
248 							pomProperties.getProperty("artifactId"), "jar", pomProperties.getProperty("version"));
249 				} catch (IOException e) {
250 					throw new KenyaException("Couldn't load pom.properties from artifact: " + jarPath, e);
251 				}
252 			}
253 		}
254 		throw new KenyaHelpException("Couldn't find pom.properties in artifact: " + jarPath);
255 	}

257 	private List<Artifact> resolveDependencies(Artifact rootArtifact, final RepositorySystem repoSystem,
258 			final RepositorySystemSession session, final List<RemoteRepository> remoteRepositories) {
259 		final CollectRequest collectRequest = new CollectRequest(new Dependency(rootArtifact, javaScope),
260 				remoteRepositories);
261 		try {
262 			final DependencyNode node = repoSystem.collectDependencies(session, collectRequest).getRoot();
263 			repoSystem.resolveDependencies(session, new DependencyRequest(node, null));
264 			final PreorderNodeListGenerator nlg = new PreorderNodeListGenerator();
265 			node.accept(nlg);
266 			rootArtifact = node.getDependency().getArtifact();
267 			// TODO: What to do with unresolved artifacts?
268 			return nlg.getArtifacts(false);
269 		} catch (DependencyCollectionException e) {
270 			throw new KenyaException("Internal error", e);
271 		} catch (DependencyResolutionException e) {
272 			throw new KenyaException("Internal error", e);
273 		}
274 	}

277 		final File userSettingsFile;
278 		if (settingsPath == null) {
279 			userSettingsFile = new File(System.getProperty(USER_HOME), M2 + File.separator + SETTINGS_XML);
280 		} else {
281 			userSettingsFile = new File(settingsPath);
282 		}
283 		if (userSettingsFile.exists()) {
284 			logger.info("Using settings file for path: {}", userSettingsFile);
285 		} else {
286 			logger.warn("Settings file does not exist: {}", settingsPath);
287 		}
288 		final DefaultSettingsBuildingRequest request = new DefaultSettingsBuildingRequest();
289 		request.setUserSettingsFile(userSettingsFile);
290 		try {
291 			final SettingsBuilder settingsBuilder = injector.getInstance(SettingsBuilder.class);
292 			return settingsBuilder.build(request).getEffectiveSettings();
293 		} catch (SettingsBuildingException e) {
294 			throw new KenyaException("Failed to load settings", e);
295 		}

298 	private final List<RemoteRepository> getRemoteRepositories(Settings settings) {
299 		final List<RemoteRepository> repositories = new ArrayList<RemoteRepository>();
300 		repositories.add(new RemoteRepository("central", "default", MAVEN_CENTRAL_REPOSITORY));
301 		@SuppressWarnings("unchecked")
302 		final Map<String, Profile> profilesAsMap = settings.getProfilesAsMap();
303 		for (String profileName : settings.getActiveProfiles()) {
304 			final Profile profile = profilesAsMap.get(profileName);
305 			for (Repository repo : profile.getRepositories()) {
306 				final RemoteRepository remoteRepository = new RemoteRepository(repo.getId(), repo.getLayout(),
307 						repo.getUrl());
308 				final RepositoryPolicy snapShotPolicy = new RepositoryPolicy(repo.getSnapshots().isEnabled(), repo
309 						.getSnapshots().getUpdatePolicy(), repo.getSnapshots().getChecksumPolicy());
310 				final RepositoryPolicy releasesPolicy = new RepositoryPolicy(repo.getReleases().isEnabled(), repo
311 						.getReleases().getUpdatePolicy(), repo.getReleases().getChecksumPolicy());
312 				remoteRepository.setPolicy(true, snapShotPolicy);
313 				remoteRepository.setPolicy(false, releasesPolicy);
314 				repositories.add(remoteRepository);
315 			}
316 		}
317 		return repositories;
318 	}

320 	private Artifact resolveRootArtifact(final RepositorySystem repoSystem, final RepositorySystemSession session,
321 			List<RemoteRepository> remoteRepositories) {
322 		final Artifact rootArtifactTemplate = getRootArtifactTemplate(jarPathProperty, artifactGavProperty);
323 		try {
324 			final ArtifactRequest artifactRequest = new ArtifactRequest(rootArtifactTemplate, remoteRepositories, null);
325 			final ArtifactResult artifactResult = repoSystem.resolveArtifact(session, artifactRequest);
326 			return artifactResult.getArtifact();
327 		} catch (ArtifactResolutionException e) {
328 			throw new KenyaException("Error resolving artifact: " + rootArtifactTemplate, e);
329 		}

356 		final ClassWorld classWorld = new ClassWorld();
357 		try {
358 			final ClassRealm realm = classWorld.newRealm(APP_REALM);
359 			for (Artifact artifact : artifacts) {
360 				realm.addURL(artifact.getFile().toURI().toURL());
361 			}
362 		} catch (DuplicateRealmException e) {
363 			throw new KenyaException("Internal error", e);
364 		} catch (MalformedURLException e) {
